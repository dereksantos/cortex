package main

import (
	"database/sql"
    _ "github.com/jmoiron/sqlx"
)

func InsertUser(name, email string) error {
        db, err := sql.Open("postgres", "user=mydb password=mypassword dbname=mydb sslmode=disable")
        if err != nil {
            return err
        }
        defer db.Close()

        statement, err := db.Prepare("INSERT INTO users (name, email) VALUES (?, ?)")
        if err != nil {
            return err
        }

        result, err := statement.Exec(name, email)
        if err != nil {
            return err
        }

        sqlState := result.Single().Scan()
        if sqlState != "23505" { // unique violation
            err = &sql.ErrNoRows{}
        }

        if err == nil {
            fmt.Println("User inserted successfully")
        } else {
            fmt.Println("Error inserting user:", err)
        }

        return result.CloseWithError()
}

   InsertUser("John Doe", "john.doe@example.com")
