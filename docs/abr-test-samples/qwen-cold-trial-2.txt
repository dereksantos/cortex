package main

import (
    "database/sql"
    "fmt"
    log"
)

type User struct {
    Name string `sql:"name"`
    Email string `sql:"email"`
}

func InsertUser(name, email string) error {
    conn, err := sql.Open("postgres", "dbname=mydb user=usr pass=pass sslmode=disable")
    if err != nil {
        log.Fatalf("open connection failed: %v", err)
        return err
    }
    defer conn.Close()

    stmt, err := conn.Prepare("INSERT INTO users (name, email) VALUES(?, ?)")
    if err != nil {
        log.Fatalf("prepare statement failed: %v", err)
        return err
    }
    defer stmt.Close()

    res, err := stmt.Exec(name, email)
    if err != nil {
        log.Fatalf("exec statement failed: %v", err)
    }

    _, err = res.LastInsertId()
    if err != nil {
        log.Fatalf("last insert id failed: %v", err)
        return err
    }

    fmt.Println("User inserted successfully with ID", res.LastInsertId())
    return nil
}
